is_valid_prostitute_trigger = {
        is_adult = yes
	is_female = yes
        is_imprisoned = no
	is_travelling = no
	is_in_army = no
	OR = {
		has_character_modifier = deviants_slave
		has_character_modifier = deviants_prostitute 
	}
	OR = {
		has_trait = deviants_succubus
		has_trait = deviants_lesser_succubus
		age < 40
	}
	NOT = { is_spouse_of = scope:actor }
	OR = {
		NOT = { has_character_modifier = deviants_fuckedsilly }
		has_character_modifier = deviants_horny
	}
}

is_valid_prostitute_partner_trigger = {
	OR = {
		AND = {
			is_male = yes
			OR = {
				has_sexuality = heterosexual
				has_sexuality = bisexual
			}
		}
		AND = {
			is_female = yes
			OR = {
				has_sexuality = homosexual
				has_sexuality = bisexual
			}
		}
	}
}
