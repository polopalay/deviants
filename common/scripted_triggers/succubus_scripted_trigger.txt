is_valid_construction_alchemy1_trigger = {
	OR = {
		any_held_title = {
			any_county_province = {
				has_building_or_higher = succubus_alchemy1
			}
		}
		has_trait_xp = {
			trait = succubus
			track = corruption
			value >= 50
		}
	}
}
