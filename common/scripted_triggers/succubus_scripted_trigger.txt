is_valid_construction_alchemy1_trigger = {
	OR = {
		any_held_title = {
			any_county_province = {
				has_building_or_higher = alchemy1
			}
		}
		has_trait_xp = {
			trait = succubus
			track = loyalty
			value >= 33
		}
	}
}

is_valid_construction_alchemy2_trigger = {
	OR = {
		any_held_title = {
			any_county_province = {
				has_building_or_higher = alchemy2
			}
		}
		has_trait_xp = {
			trait = succubus
			track = corruption
			value >= 50
		}
	}
}

is_valid_construction_alchemy3_trigger = {
	OR = {
		any_held_title = {
			any_county_province = {
				has_building_or_higher = alchemy3
			}
		}
		AND = {
			has_trait_xp = {
				trait = succubus
				track = corruption
				value >= 50
			}
			has_trait_xp = {
				trait = succubus
				track = loyalty
				value >= 66
			}
		}

	}

}
