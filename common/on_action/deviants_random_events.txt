quarterly_playable_pulse = {
	trigger = {
		is_adult = yes
		has_trait = deviants_succubus
		is_incapable = no
		is_imprisoned = no
		is_travelling = no
		is_in_army = no
		is_landed = yes
		is_female = yes
		exists = capital_province
		NOT = { has_trait = celibate }
	}
	random_on_action = {
	 	chance_of_no_event = 4
		1 = deviants_courtier_on_action
		1 = deviants_devil_on_action
		1 = deviants_loyalty_on_action
		1 = deviants_blackmail_on_action
	}
}
