carn_force_start_prostitution_interaction = {
	icon = start_prostitution_interaction
	desc = carn_force_start_prostitution_interaction.desc
	extra_icon = hook_icon
	category = interaction_category_prison
	use_diplomatic_range = yes
	common_interaction = yes

	is_shown = {
		scope:actor = {
			is_adult = yes
			is_ruler = yes
			faith = {
				has_doctrine = doctrine_adultery_men_accepted
				has_doctrine = doctrine_adultery_women_accepted
			}
		}
		scope:recipient = {
			is_adult = yes
			is_female = yes
			OR = {
				age < carn_prostitution_age_cutoff
				is_immortal = yes
			}
			NOT = { has_trait = lifestyle_prostitute }
			OR = {
				is_courtier_of = scope:actor
				scope:actor = scope:recipient
				AND = {
					is_ruler = no
					is_imprisoned_by = scope:actor
				}
			}
		}
	}

	is_valid = {
		scope:recipient = {
			OR = {
				scope:actor = scope:recipient
				AND = {
					OR = {
						is_courtier_of = scope:actor
						is_imprisoned_by = scope:actor
					}
					scope:actor = {
						any_courtier = {
							has_trait = lifestyle_prostitute
							count < maximum_prostitute
						}
					}
				}
			}
		}
	}

	on_accept = {
		scope:recipient = {
			if = {
				limit = { has_trait = celibate }
				remove_trait = celibate
			}
			add_trait = lifestyle_prostitute
			if = {
				limit = { is_imprisoned_by = scope:actor }
				release_from_prison = yes
			}
		}
		scope:actor = {
			if = {
				limit = {
					NOT = { is_liege_or_above_of = scope:recipient }
					NOT = { scope:recipient = scope:actor }
				}
				add_courtier = scope:recipient
			}
			if = {
				limit = {
					scope:hook = yes
					has_hook = scope:recipient
				}
				use_hook = scope:recipient
			}
		}
	}

	auto_accept = {
		OR = {
			scope:recipient = { is_imprisoned_by = scope:actor }
			scope:recipient = scope:actor
			scope:recipient = { has_trait = slave }
			custom_description = {
				text = "spending_hook"
				subject = scope:actor
				object = scope:recipient
				scope:hook = yes
			}
		}
	}

	send_option = {
		is_valid = {
			scope:actor = {
				has_usable_hook = scope:recipient
			}
		}
		flag = hook
		localization = GENERIC_SPEND_A_HOOK
	}
	send_options_exclusive = no

	ai_accept = {
		base = -50
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient
			multiplier = 1
		}
		modifier = { add = -50 scope:recipient = { has_trait = chaste } }
	}

	ai_frequency = 2
	ai_targets = {
		ai_recipients = courtiers
		ai_recipients = prisoners
		ai_recipients = vassals
		ai_recipients = self
	}
	ai_potential = { always = yes }
	ai_will_do = {
		base = 0
		modifier = { add = 25 scope:actor = { faith = { has_doctrine = tenet_polyamory } } }
		modifier = { add = -25 scope:actor = { has_trait = chaste } }
		modifier = { add = 5 scope:actor = { has_trait = lustful } }
		modifier = { add = 5 scope:actor = { has_trait = rakish } }
		modifier = { add = 5 scope:actor = { has_trait = greedy } }
	}
}
