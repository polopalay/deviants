deviants_absorb_interaction = {
	icon = icon_shift_privileges
	desc = deviants_absorb_interaction.desc
    	category = interaction_category_vassal
    	use_diplomatic_range = yes
	common_interaction = yes
	cooldown = { months = 1 }
	is_shown = {
		scope:actor = {
			has_trait = deviants_succubus
			is_ruler = yes
			NOT = { is_at_war_with = scope:recipient }
			has_trait_xp = { trait = deviants_succubus track = loyalty value < 100 }
		}
		scope:recipient = {
			OR = {
				is_courtier_of = scope:actor
				is_vassal_of = scope:actor
			}
			has_trait_xp = { trait = deviants_lesser_succubus value >= 100 }
		}
	}
	on_accept = {
		scope:recipient = {
			add_character_modifier = deviants_buff
			add_trait_xp = {
				trait = deviants_lesser_succubus
				value = -100
			}
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = loyalty
				value = 1
			}
		}
	}
	auto_accept = yes
	ai_frequency = 12
	ai_targets = { 
		ai_recipients = courtiers 
		ai_recipients = prisoners
		ai_recipients = vassals
	}
	ai_potential = { always = yes }
	ai_will_do = { base = 100 }
}

deviants_cultist_interaction = {
	icon = ask_for_conversion_interaction
	desc = deviants_cultist_interaction.desc
    	category = interaction_category_vassal
    	use_diplomatic_range = yes
	common_interaction = yes
	cooldown = { years = 1 }

	is_shown = {
		scope:actor = {
			has_trait = deviants_succubus
			is_ruler = yes
		}
		scope:recipient = {
			is_ruler = yes
			primary_title.tier >= tier_county
		}
	}
	is_valid = {
		scope:recipient = {
			OR = {
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				AND = {
					scope:actor = { has_trait_xp = { trait = deviants_succubus track = loyalty value >= 5 } }
					primary_title.tier = tier_county
				}
				AND = {
					scope:actor = { has_trait_xp = { trait = deviants_succubus track = loyalty value >= 10 } }
					primary_title.tier = tier_duchy
				}
				AND = {
					scope:actor = { has_trait_xp = { trait = deviants_succubus track = loyalty value >= 25 } }
					primary_title.tier = tier_kingdom
				}
				AND = {
					scope:actor = { has_trait_xp = { trait = deviants_succubus track = loyalty value >= 100 } }
					primary_title.tier = tier_empire
				}
			}
		}
	}

	send_option = {
		flag = lilith	
		localization = lilith
	}

	send_option = {
		flag = satan	
		localization = satan
	}

	on_accept = {
		if = {
			limit = {
				scope:lilith = yes
			}
			scope:recipient = {
				set_character_faith_with_conversion = faith:deviants_lilith
			}
		}
		if = {
			limit = {
				scope:satan = yes
			}
			scope:recipient = {
				set_character_faith_with_conversion = faith:deviants_satan
			}
		}
	}

	auto_accept = yes
}
