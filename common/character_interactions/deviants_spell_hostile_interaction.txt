deviants_steal_gold_interaction = {
	icon = dgold
	desc = deviants_steal_gold_interaction.desc
    	category = interaction_category_hostile
    	use_diplomatic_range = yes
	cooldown = { years = 1 }
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			has_trait_xp = {
				trait = deviants_succubus
				track = power
				value >= 1
			} 
		}
		scope:recipient = {
			gold >= 100
			NOR = {
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
			}
		}
	}
	on_accept = {
		scope:recipient = {
			pay_short_term_gold = {
				target = scope:actor
				gold = 20
			}
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = power
				value = -1
			}
		}
	}
	auto_accept = yes
	ai_frequency = 12
	ai_targets = { 
		ai_recipients = liege
		ai_recipients = vassals
		ai_recipients = peer_vassals
		ai_recipients = neighboring_rulers
	}
	ai_potential = { always = yes }
	ai_will_do = { 
		base = 5 
		modifier = {
			add = 5
			scope:actor = {
				OR = {
					has_relation_rival = scope:recipient
					has_relation_nemesis = scope:recipient
				}
			}
		}
	}
}

deviants_rumor_interaction = {
	icon = drumor
	desc = deviants_rumor_interaction.desc
    	category = interaction_category_hostile
    	use_diplomatic_range = yes
	cooldown = { years = 1 }
	is_shown = {
		scope:actor = {
			any_held_title = {
				any_county_province = {
					has_building_or_higher = deviants_brothel1 
				}
			}
			has_trait_xp = {
				trait = deviants_succubus
				track = power
				value >= 2
			} 
		}
		scope:recipient = {
			OR = {
				scope:actor = scope:recipient
				NOR = {
					has_trait = deviants_succubus
					has_trait = deviants_lesser_succubus
				}
			}
		}
	}
	is_valid = {
		scope:actor = {
			any_held_title = {
				any_county_province = {
					has_building_or_higher = deviants_brothel1 
				}
			}
		}
	}
	on_accept = {
		scope:recipient = {
			random_list = {
				1 = {
					give_nickname = nick_the_lewd
				}
				1 = {
					give_nickname = nick_the_lecher
				}
				1 = {
					give_nickname = nick_the_bewitched
				}
			}
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = power
				value = -2
			}
		}
	}
	auto_accept = yes
	ai_frequency = 60
	ai_targets = { ai_recipients = scripted_relations }
	ai_potential = { always = yes }
	ai_will_do = { 
		base = 0 
		modifier = {
			add = 100
			scope:actor = scope:recipient
			scope:actor = {
				NOT = {
					has_nickname = nick_the_lewd
					has_nickname = nick_the_lecher
					has_nickname = nick_the_bewitched
				}
			}
		}
		modifier = {
			add = 5
			scope:actor = {
				OR = {
					has_relation_rival = scope:recipient
					has_relation_nemesis = scope:recipient
				}
			}
		}
	}
}
