deviants_steal_gold_interaction = {
	icon = icon_gold
	desc = deviants_steal_gold_interaction.desc
    	category = interaction_category_hostile
    	use_diplomatic_range = yes
	common_interaction = yes
	cooldown = { years = 1 }
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			has_trait_xp = {
				trait = deviants_succubus
				track = power
				value >= 10
			} 
		}
		scope:recipient = {
			NOR = {
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
			}
		}
	}
	on_accept = {
		scope:recipient = {
			pay_short_term_gold = {
				target = scope:actor
				gold = 200
			}
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = power
				value = -10
			}
		}
	}
	is_valid = {
		scope:actor = {
			is_valid_construction_alchemy1_trigger = yes
		}
	}
	auto_accept = yes
	ai_frequency = 12
	ai_targets = {
		ai_recipients = liege
		ai_recipients = vassals
		ai_recipients = peer_vassals
	}
	ai_potential = { always = yes }
	ai_will_do = { 
		base = 5 
	}
}
