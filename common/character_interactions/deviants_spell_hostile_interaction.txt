deviants_mind_break_interaction = {
	icon = titsbad
	common_interaction = no
	desc = deviants_mind_break_interaction.desc
    	category = interaction_category_hostile
    	use_diplomatic_range = yes
	cooldown = { years = 5 }
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:recipient = {
			is_adult = yes
			age < 40
		}
		scope:actor = {
			has_trait_xp = {
				trait = deviants_succubus
				track = demonpower
				value >= 20
			} 
		}
	}
	on_accept = {
		scope:recipient = {
			remove_trait = chaste
			add_trait = lustful	
			if = {
				limit = {
					is_female = yes
				}
				if = {
					limit = {
						NOT = { has_character_modifier = deviants_prostitute }
					}
					add_character_modifier = deviants_prostitute
				}
				if = {
					limit = { has_character_modifier = deviants_prostitute }
					remove_character_modifier = deviants_prostitute
					add_character_modifier = deviants_public_toilet
				}
			}
			add_secret = { 
				type = secret_deviant
			}
			add_secret = { 
				type = secret_witch
			}
			add_secret = { 
				type = secret_cannibal
			}
			random_list = {
				25 = {
					give_nickname = nick_the_lewd
				}
				25 = {
					give_nickname = nick_the_lecher
				}
				15 = {
					give_nickname = nick_the_witch
				}
				15 = {
					give_nickname = nick_the_bewitched
				}
				10 = {
					give_nickname = nick_the_cannibal
				}
				10 = {
					add_character_modifier = diplomacy_majesty_4090_no_raiment_modifier
				}
			}
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = demonpower
				value = -20
			}
		}
	}
	auto_accept = yes
	ai_frequency = 12
	ai_targets = { ai_recipients = scripted_relations }
	ai_potential = { always = yes }
	ai_will_do = { 
		base = 0 
		modifier = {
			add = 10
			scope:actor = {
				OR = {
					has_relation_rival = scope:recipient
					has_relation_nemesis = scope:recipient
				}
			}
		}
	}
}
