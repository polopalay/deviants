naked_interaction = {
	desc = naked_interaction.desc
	auto_accept = yes
	use_diplomatic_range = yes
	ignores_pending_interaction_block = yes
	category = interaction_category_friendly
	is_shown = {
		OR = {
			scope:actor = scope:recipient
			scope:recipient = { is_courtier_of = scope:actor }
		}
		scope:actor = {
			NOT = { has_trait = succubus }
		}
		scope:recipient = {
			has_trait = lifestyle_prostitute
			NOT = { has_character_modifier = diplomacy_majesty_4090_no_raiment_modifier }
		}
	}
	on_accept = {
		scope:recipient = {
			add_character_modifier = diplomacy_majesty_4090_no_raiment_modifier
		}
	}
	ai_targets = { ai_recipients = self }
	ai_potential = { always = yes }
	ai_frequency = 1
	ai_will_do = { base = 5 }
}
