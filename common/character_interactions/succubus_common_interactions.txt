succubus_switch_body_interaction = {
	icon = two_body
	desc = succubus_switch_body_interaction.desc
	category = interaction_category_diplomacy
	interface_priority = 1
	common_interaction = yes
	use_diplomatic_range = yes
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = {
			is_ai = no
			OR = {
				has_trait = succubus
				has_trait = lesser_succubus
			}
		}
		scope:recipient = {
			is_ruler = yes
			primary_title.tier >= tier_county
			OR = {
				has_trait = succubus
				has_trait = lesser_succubus
				has_trait = deviant_virgin
			}
		}
	}
	on_accept = {
		scope:actor = { set_player_character = scope:recipient }
	}
	auto_accept = yes
}

naked_interaction = {
	desc = naked_interaction.desc
	auto_accept = yes
	use_diplomatic_range = yes
	ignores_pending_interaction_block = yes
	category = interaction_category_friendly
	is_shown = {
		OR = {
			scope:actor = scope:recipient
			scope:recipient = { is_courtier_of = scope:actor }
		}
		scope:actor = {
			NOT = { has_trait = succubus }
		}
		scope:recipient = {
			has_trait = lifestyle_prostitute
			NOT = { has_character_modifier = diplomacy_majesty_4090_no_raiment_modifier }
		}
	}
	on_accept = {
		scope:recipient = {
			add_character_modifier = diplomacy_majesty_4090_no_raiment_modifier
		}
	}
	ai_targets = { ai_recipients = self }
	ai_potential = { always = yes }
	ai_frequency = 1
	ai_will_do = { base = 5 }
}
