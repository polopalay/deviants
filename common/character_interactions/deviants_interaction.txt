deviants_prostitute_interaction = {
	icon = d_sex
	desc = deviants_prostitute_interaction.desc
    	category = interaction_category_diplomacy
    	use_diplomatic_range = yes
	common_interaction = yes

    	is_shown = {
		NOT = { scope:recipient = scope:actor }
        	scope:actor = {
            		is_adult = yes
            		is_incapable = no
            		is_imprisoned = no
			is_travelling = no
			is_in_army = no
			NOT = { is_at_war_with = scope:recipient }
        	}
        	scope:recipient = {
            		is_adult = yes
            		is_incapable = no
            		is_imprisoned = no
			is_travelling = no
			is_in_army = no
			OR = {
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				AND = {
					has_character_modifier = deviants_prostitute
					age < 40
				}
			}
			NOT = { any_relation = { type = deviants_slave_owner } }
        	}
    	}

	
	is_valid = {
		scope:recipient = {
			OR = {
				is_valid_prostitute_trigger = yes
				any_vassal = {
					is_valid_prostitute_trigger = yes
					scope:recipient = { is_valid_construction_brothel1_trigger = yes }
				}
				any_courtier = {
					is_valid_prostitute_trigger = yes
					scope:recipient = { is_valid_construction_brothel1_trigger = yes }
				}
				any_relation = {
					type = deviants_partner
					is_valid_prostitute_trigger = yes
					NOT = { 
						has_trait =deviants_succubus
					}
					scope:recipient = { is_valid_construction_brothel2_trigger = yes }
				}
				any_relation = {
					type = deviants_slaver
					is_valid_prostitute_trigger = yes
					NOT = { 
						has_trait =deviants_succubus
					}
					scope:recipient = { is_valid_construction_brothel3_trigger = yes }
				}
			}
		}
	}

	populate_recipient_list = {
		scope:recipient = {
			if = {
				limit = { is_valid_prostitute_trigger = yes }
				add_to_list = characters
			}
			every_courtier = {
				limit = { 
					is_valid_prostitute_trigger = yes 
					scope:recipient = { is_valid_construction_brothel1_trigger = yes }
				}
				add_to_list = characters
			}
			every_vassal = {
				limit = { 
					is_valid_prostitute_trigger = yes
					scope:recipient = { is_valid_construction_brothel1_trigger = yes }
				}
				add_to_list = characters
			}
			every_relation = {
				type = deviants_partner
				limit = { 
					is_valid_prostitute_trigger = yes
					NOT = { 
						has_trait =deviants_succubus
					}
					scope:recipient = { is_valid_construction_brothel2_trigger = yes }
				}
				add_to_list = characters
			}
			every_relation = {
				type = deviants_slaver
				limit = { 
					is_valid_prostitute_trigger = yes 
					NOT = { 
						has_trait =deviants_succubus
					}
					scope:recipient = { is_valid_construction_brothel3_trigger = yes }
				}
				add_to_list = characters
			}
		}
	}

    	on_accept = {
		scope:secondary_recipient = {
			add_character_modifier = { modifier = deviants_fuckedsilly months = 2 }
			had_sex_with_effect = { CHARACTER = scope:actor PREGNANCY_CHANCE = pregnancy_chance }
		}
        	scope:actor = {
			add_opinion = {
				target = scope:recipient
				modifier = have_sex_opinion
			}
			if = {
				limit = { 
					scope:secondary_recipient = { has_character_modifier = deviants_prostitute }
				}
				if = {
					limit = { scope:secondary_recipient = scope:recipient }
					pay_short_term_gold = { target = scope:recipient gold = 5 } 
				}
				else = {
					pay_short_term_gold = { target = scope:recipient gold = 2 }
					pay_short_term_gold = { target = scope:secondary_recipient gold = 3 }
				}
				random = {
					chance = 10
					modifier = { add = 40 has_trait = generous }
					modifier = { add = 5 scope:secondary_recipient = { has_trait = beauty_good_1 } }
					modifier = { add = 8 scope:secondary_recipient = { has_trait = beauty_good_2 } }
					modifier = { add = 10 scope:secondary_recipient = { has_trait = beauty_good_3 } }
					pay_short_term_gold = { target = scope:secondary_recipient gold = 1 }
				}
			}
    		}
		if = {
			limit = { scope:secondary_recipient = { has_trait = deviants_succubus } }
			scope:recipient = {
				add_trait_xp = { trait = deviants_succubus track = power value = 1 }
			}
		}
		if = {
			limit = {
				NOT = { scope:secondary_recipient = scope:recipient }
				scope:recipient = { has_trait = deviants_succubus } 
			}
			random = {
				chance = 25
				scope:recipient = {
					add_trait_xp = { trait = deviants_succubus track = corruption value = 1 }
				}
			}
		}
		if = {
			limit = { scope:secondary_recipient = { has_trait = deviants_lesser_succubus } }
				scope:secondary_recipient = {
					add_trait_xp = { trait = deviants_lesser_succubus value = 2 }
				}
    		}
	}

	auto_accept = {
		scope:secondary_recipient = {
			OR = {
				scope:secondary_recipient = scope:recipient
				is_courtier_of = scope:recipient
				is_vassal_of = scope:recipient
				has_relation_deviants_slave_owner = scope:recipient
				has_trait = deviants_lesser_succubus
				has_trait = deviants_succubus
			}
		}
	}

	ai_accept = {
		base = 10
		modifier = {
			add = -50
			scope:secondary_recipient = { 
				OR = {
					is_close_family_of = scope:actor
					is_child_of = scope:actor
				}
			}
		}
		modifier = {
			add = 100
			scope:secondary_recipient = { 
				OR = {
					has_trait = deviants_lesser_succubus
					has_trait = deviants_succubus
				}
			}
		}
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:secondary_recipient
			multiplier = 1
		}
	}

	ai_frequency = 12
	ai_targets = {
		ai_recipients = liege
		ai_recipients = vassals
		ai_recipients = peer_vassals
		ai_recipients = neighboring_rulers
		ai_recipients = realm_characters
	}
	ai_potential = { 
		primary_title.tier >= tier_county
	}
	ai_will_do = {
		base = 0
		modifier = {
			add = 5
			scope:actor = {
				gold > 10
			}
		}
				modifier = {
		add = 5
			scope:actor = {
				has_trait = lustful
			}
		}
		modifier = {
			add = -10
			scope:secondary_recipient = {
				is_close_family_of = scope:actor
			}
		}
		modifier = {
			add = 5
			scope:secondary_recipient = {
				OR = {
					has_trait = physique_good_1
					has_trait = physique_good_2
					has_trait = physique_good_3
				}
			}
		}
		modifier = {
			add = -5
			scope:secondary_recipient = {
				OR = {
					has_trait = physique_bad_1
					has_trait = physique_bad_2
					has_trait = physique_bad_3
				}
			}
		}
		modifier = {
			add = 5
			scope:secondary_recipient = {
				OR = {
					has_trait = intellect_good_1
					has_trait = intellect_good_2
					has_trait = intellect_good_3
				}
			}
		}
		modifier = {
			add = -5
			scope:secondary_recipient = {
				OR = {
					has_trait = intellect_bad_1
					has_trait = intellect_bad_2
					has_trait = intellect_bad_3
				}
			}
		}
		modifier = {
			add = -5
			scope:secondary_recipient = {
				has_trait = beauty_bad_1
			}
		}
		modifier = {
			add = -8
			scope:secondary_recipient = {
				has_trait = beauty_bad_2
			}
		}
		modifier = {
			add = -10
			scope:secondary_recipient = {
				has_trait = beauty_bad_3
			}
		}
		modifier = {
			add = 5
			scope:secondary_recipient = {
				has_trait = beauty_good_1
			}
		}
		modifier = {
			add = 8
			scope:secondary_recipient = {
				has_trait = beauty_good_2
			}
		}
		modifier = {
			add = 10

				has_trait = beauty_good_3

		}
		modifier = {

		}
	}

deviants_become_prostitute_interaction = {
	icon = d_start_prostitution
	desc = deviants_become_prostitute_interaction.desc
    	category = interaction_category_diplomacy
    	use_diplomatic_range = yes
	common_interaction = yes
	cooldown = { months = 1 }

    	is_shown = {
        	scope:actor = {
            		is_adult = yes
			is_landed = yes
			has_trait = deviants_succubus
			NOT = { is_at_war_with = scope:recipient }
        	}
        	scope:recipient = {
            		is_adult = yes
			is_female = yes
			NOT = { has_character_modifier = deviants_prostitute }
			OR = {
				is_courtier_of = scope:actor
				is_vassal_of = scope:actor
				scope:actor = scope:recipient
				AND = {
					is_ruler = no
					is_imprisoned_by = scope:actor
				}
			}
        	}
    	}

	is_valid = {
		scope:recipient = {
			OR = {
				AND = {
					is_vassal_of = scope:actor
					scope:actor = {
						any_vassal = {
							has_character_modifier = deviants_prostitute
							count < 3
						}	
					}
				}
				AND = {
					OR = {
						is_courtier_of = scope:actor
						is_imprisoned_by = scope:actor
					}
					scope:actor = {
						any_courtier = {
							has_character_modifier = deviants_prostitute
							count < 3
						}	
					}
				}
			}
		}
	}

    	on_accept = {
		scope:recipient = {
			add_character_modifier = diplomacy_majesty_4090_no_raiment_modifier
			add_character_modifier = deviants_prostitute
			add_character_flag = blocked_from_leaving
			if = {
				limit = { is_imprisoned_by = scope:actor }
				release_from_prison = yes
			}
			if = {
				limit = {
					scope:recipient = {
						is_spouse_of = scope:actor
						exposed_cheating_on_spouse_trigger = { SPOUSE = scope:actor }
					}
					every_spouse = {
						divorce = scope:recipient
					}
				}
			}
		}
		scope:actor = {
			if = {
				limit = { 
					NOT = { is_liege_or_above_of = scope:recipient } 
					NOT = { scope:recipient = scope:actor }
				}
				add_courtier = scope:recipient
			}
			if = {
				limit = {
					scope:hook = yes
					has_hook = scope:recipient
				}
				use_hook = scope:recipient
			}
		}
    	}

	auto_accept = {
		OR = {
			scope:recipient = { is_imprisoned_by = scope:actor }
			scope:recipient = scope:actor
			custom_description = {
				text = "spending_hook"
				subject = scope:actor
				object = scope:recipient
				scope:hook = yes
			}
		}
	}

	send_option = {
		is_valid = {
			scope:actor = {
				has_usable_hook = scope:recipient
			}
		}
		flag = hook	
		localization = GENERIC_SPEND_A_HOOK
	}
	send_options_exclusive = no

	ai_accept = {
		base = -50
		modifier = {
			add = 100
			scope:recipient = { has_trait = deviants_lesser_succubus }
		}
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient
			multiplier = 1
		}
	}

	ai_frequency = 12
	ai_targets = { 
		ai_recipients = courtiers 
		ai_recipients = prisoners
		ai_recipients = vassals
	}
	ai_potential = { always = yes }
	ai_will_do = {
		base = 50
		modifier = {
			add = 50
			scope:actor = scope:recipient
		}
	}
}

    	}

	auto_accept = yes

	ai_frequency = 12
