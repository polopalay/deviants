deviants_nude_interaction = {
	icon = undress_interaction
	desc = deviants_nude_interaction.desc
    	category = interaction_category_hostile
    	use_diplomatic_range = yes
	cooldown = { years = 1 }
	is_shown = {
		OR = {
			AND = {
				scope:actor = {
					has_trait_xp = {
						trait = deviants_succubus
						track = daring
						value >= 10
					} 
				}
				scope:recipient = {
					is_adult = yes
					is_female = yes
					NOR = {
						has_trait = deviants_succubus
						has_trait=deviants_lesser_succubus
					}
					NOT = { has_character_modifier = diplomacy_majesty_4090_no_raiment_modifier }
				}
			}
			AND = {
				scope:actor = {
					has_trait_xp = {
						trait = deviants_succubus
						track = daring
						value >= 1
					} 
				}
				scope:recipient = {
					is_adult = yes
					is_female = yes
					OR = {
						has_trait = deviants_succubus
						has_trait=deviants_lesser_succubus
					}
					NOT = { has_character_modifier = diplomacy_majesty_4090_no_raiment_modifier }
				}
			}
		}
	}
	on_accept = {
		scope:recipient = {
			add_character_modifier = diplomacy_majesty_4090_no_raiment_modifier
			random_courtier = {
				limit = { is_female = no is_adult = yes }
				had_sex_with_effect = { CHARACTER = scope:recipient PREGNANCY_CHANCE = pregnancy_chance } 
				if = {
					limit = { scope:recipient = { has_trait = deviants_succubus } }
					scope:recipient = { 
						add_character_modifier = { modifier = deviants_fuckedsilly months = 1 } 
						add_trait_xp = { trait = deviants_succubus track = power value = 1 }
					}
				}
			}
		}
		scope:actor = {
			if = {
				limit = {
					NOR = {
						has_trait = deviants_succubus
						has_trait=deviants_lesser_succubus
					}
				}
				add_trait_xp = {
					trait = deviants_succubus
					track = daring
					value = -10
				}
			}
			else = {
				add_trait_xp = {
					trait = deviants_succubus
					track = daring
					value = -1
				}
			}
		}
	}
	auto_accept = yes
}

deviants_shake_mind_interaction = {
	icon = d_fascinate
	desc = deviants_mind_break_interaction.desc
    	category = interaction_category_hostile
    	use_diplomatic_range = yes
	cooldown = { years = 5 }
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:recipient = {
			is_adult = yes
			is_ruler = yes
			is_female = yes
			age < 40
			NOT  = { 
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
			}
		}
		scope:actor = {
			has_trait_xp = {
				trait = deviants_succubus
				track = daring
				value >= 2
			} 
		}
	}
	is_valid = {
		scope:actor = { is_valid_construction_alchemy2_trigger = yes }
	}
	on_accept = {
		scope:recipient = {
			add_character_modifier = deviants_prostitute
			random_courtier = {
				limit = { is_female = no is_adult = yes }
				had_sex_with_effect = { CHARACTER = scope:recipient PREGNANCY_CHANCE = pregnancy_chance }
				scope:recipient = { add_character_modifier = { modifier = deviants_fuckedsilly months = 1 } }
				had_sex_with_effect = { CHARACTER = scope:actor PREGNANCY_CHANCE = pregnancy_chance } 
				if = {
					limit = { is_knight = yes }
					scope:actor = {
						random = {
							chance = 25
							add_trait_xp = { trait = deviants_succubus track = loyalty value = 1 } 
						}
					} 
				}
				else = {
					scope:actor = { 
						random = {
							chance = 25
							add_trait_xp = { trait = deviants_succubus track = corruption value = 1 } 
						}
					} 
				}
			}
			add_character_modifier = { modifier = lost_in_lust_modifier months = 4 }
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = daring
				value = -2
			}
		}
	}
	auto_accept = yes
}
