deviants_offer_orgy = {
	icon = d_sway
	desc = deviants_offer_orgy.desc
    	category = interaction_category_diplomacy
    	use_diplomatic_range = yes
	common_interaction = yes
	cooldown = { years = 5 }
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:recipient = {
			is_adult = yes
			is_ruler = yes
			NOT = { is_at_war_with = scope:actor }
			NOT  = { is_spouse_of = scope:actor }
		}
		scope:actor = {
			has_trait = deviants_succubus
			is_adult = yes
		}
	}
	is_valid = {
		scope:actor = {
			any_held_title = {
				any_county_province = {
					has_building_or_higher = deviants_brothel1 
				}
			}
		}
	}
	on_accept = {
		scope:recipient = {
			if = {
				limit = { is_female = no }
				had_sex_with_effect = { CHARACTER = scope:actor PREGNANCY_CHANCE = pregnancy_chance }
				scope:actor = { add_character_modifier = { modifier = deviants_fuckedsilly months = 1 } }
				scope:actor = { add_trait_xp = { trait = deviants_succubus track = daring value = 1 } } 
			}
			every_courtier_or_guest = {
				limit = { is_adult = yes is_female = no }
				had_sex_with_effect = { CHARACTER = scope:actor PREGNANCY_CHANCE = pregnancy_chance }
				scope:actor = { 
					add_character_modifier = { modifier = deviants_fuckedsilly months = 1 }
					add_trait_xp = { trait = deviants_succubus track = daring value = 1 }
				}
			}
		}
	}
	ai_accept = {
		base = 20
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient
			multiplier = 1
		}
	}
	ai_frequency = 24
	ai_targets = { 	
		ai_recipients = liege
		ai_recipients = vassals
		ai_recipients = peer_vassals
		ai_recipients = neighboring_rulers
	}
	ai_potential = { always = yes }
	ai_will_do = { base = 100 }
}

deviants_nude_interaction = {
	icon = d_undress
	desc = deviants_nude_interaction.desc
    	category = interaction_category_hostile
    	use_diplomatic_range = yes
	cooldown = { years = 1 }
	is_shown = {
		scope:actor = {
			has_trait_xp = {
				trait = deviants_succubus
				track = daring
				value >= 5
			} 
		}
		scope:recipient = {
			is_adult = yes
			is_female = yes
			NOT = { has_character_modifier = diplomacy_majesty_4090_no_raiment_modifier }
		}
	}
	is_valid = {
		scope:actor = {
			any_held_title = {
				any_county_province = {
					has_building_or_higher = deviants_brothel1 
				}
			}
		}
	}
	on_accept = {
		scope:recipient = {
			add_character_modifier = { modifier = deviants_prostitute years = 1 }
			add_character_modifier = diplomacy_majesty_4090_no_raiment_modifier
			random_courtier = {
				limit = { is_female = no is_knight = no is_adult = yes }
				had_sex_with_effect = { CHARACTER = scope:recipient PREGNANCY_CHANCE = pregnancy_chance } 
				if = {
					limit = { NOT = { scope:actor = scope:recipient } }
					had_sex_with_effect = { CHARACTER = scope:actor PREGNANCY_CHANCE = pregnancy_chance } 
					scope:actor = { 
						add_trait_xp = { trait = deviants_succubus track = power value = 1 } 
						add_character_modifier = { modifier = deviants_fuckedsilly months = 1 }
					}
				}
			}
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = daring
				value = -5
			}
		}
	}
	auto_accept = yes
	ai_frequency = 24
	ai_targets = { ai_recipients = self }
	ai_potential = { always = yes }
	ai_will_do = { base = 10 }
}

deviants_mind_break_interaction = {
	icon = d_mesmerize
	desc = deviants_mind_break_interaction.desc
    	category = interaction_category_hostile
    	use_diplomatic_range = yes
	cooldown = { years = 5 }
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:recipient = {
			is_adult = yes
			is_ruler = yes
			is_female = yes
			NOT  = { 
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				has_character_modifier = deviants_slave
			}
		}
		scope:actor = {
			has_trait_xp = {
				trait = deviants_succubus
				track = daring
				value >= 20
			} 
		}
	}
	is_valid = {
		scope:actor = {
			any_held_title = {
				any_county_province = {
					has_building_or_higher = deviants_brothel3 
				}
			}
		}
	}
	on_accept = {
		scope:recipient = {
			remove_trait = chaste
			add_trait = lustful	
			add_character_modifier = { modifier = deviants_slave }
			add_character_modifier = { modifier = deviants_prostitute }
			every_courtier_or_guest = {
				limit = { is_adult = yes is_female = no }
				had_sex_with_effect = { CHARACTER = scope:recipient PREGNANCY_CHANCE = pregnancy_chance }
				had_sex_with_effect = { CHARACTER = scope:actor PREGNANCY_CHANCE = pregnancy_chance } 
				scope:actor = { 
					add_trait_xp = { trait = deviants_succubus track = power value = 1 } 
				} 
				if = {
					limit = { is_knight = yes }
					scope:actor = { 
						add_trait_xp = { trait = deviants_succubus track = loyalty value = 1 } 
					} 
				}
				else = {
					scope:actor = { 
						add_trait_xp = { trait = deviants_succubus track = power value = 1 } 
					} 
				}
				scope:recipient = { 
					add_character_modifier = { modifier = deviants_fuckedsilly months = 1 } 
				}
			}
			add_character_modifier = { modifier = lost_in_lust_modifier months = 4 }
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = daring
				value = -20
			}
		}
	}
	auto_accept = yes
	ai_frequency = 12
	ai_targets = { 	
		ai_recipients = liege
		ai_recipients = vassals
		ai_recipients = peer_vassals
		ai_recipients = neighboring_rulers
	}
	ai_potential = { always = yes }
	ai_will_do = { 
		base = 0 
		modifier = {
			add = 5
			scope:actor = {
				OR = {
					has_relation_rival = scope:recipient
					has_relation_nemesis = scope:recipient
				}
			}
		}
	}
}
