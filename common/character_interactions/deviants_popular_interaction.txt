deviants_offer_orgy = {
	icon = deviantslove
	desc = deviants_offer_orgy.desc
    	category = interaction_category_diplomacy
    	use_diplomatic_range = yes
	cooldown = { years = 2 }
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:recipient = {
			NOT = { is_at_war_with = scope:actor }
			is_adult = yes
			is_ruler = yes
			NOT  = { is_spouse_of = scope:actor }
		}
		scope:actor = {
			has_trait = deviants_succubus
			is_adult = yes
			any_held_title = {
				any_county_province = {
					has_building_or_higher = deviants_brothel1 
				}
			}
		}
	}
	on_accept = {
		scope:recipient = {
			if = {
				limit = { is_female = no }
				had_sex_with_effect = { CHARACTER = scope:actor PREGNANCY_CHANCE = 0 }
				scope:actor = { add_character_modifier = { modifier = deviants_fuckedsilly months = 1 } }
				scope:actor = { add_trait_xp = { trait = deviants_succubus track = daring value = 1 } } 
			}
			every_courtier_or_guest = {
				limit = { is_adult = yes is_female = no }
				had_sex_with_effect = { CHARACTER = scope:actor PREGNANCY_CHANCE = 0 }
				scope:actor = { 
					add_character_modifier = { modifier = deviants_fuckedsilly months = 1 }
					add_trait_xp = { trait = deviants_succubus track = daring value = 1 }
				}
			}
		}
	}
	ai_accept = {
		base = 20
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient
			multiplier = 1
		}
	}
	ai_frequency = 24
	ai_targets = { 	
		ai_recipients = liege
		ai_recipients = vassals
		ai_recipients = peer_vassals
		ai_recipients = neighboring_rulers
	}
	ai_potential = { always = yes }
	ai_will_do = { base = 100 }
}

deviants_dispel_interaction = {
	icon = deviantsmagic
	desc = deviants_dispel_interaction.desc
    	category = interaction_category_diplomacy
    	use_diplomatic_range = yes
	cooldown = { years = 1 }
	cooldown_against_recipient = { months = 2 }
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			is_adult = yes
			is_ruler = yes
			is_female = yes
			NOT  = { 
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
			}
			OR = {
				has_character_modifier = deviants_prostitute
				has_character_modifier = deviants_slave
			}
		}
		scope:recipient = {
			is_ruler = yes
			NOT = { is_at_war_with = scope:actor }
			has_trait = deviants_succubus 
		}
	}
	send_option = {
		is_valid = {
			scope:actor = {
				has_usable_hook = scope:recipient
			}
		}
		flag = hook	
		localization = GENERIC_SPEND_A_HOOK
	}
	send_options_exclusive = no
	auto_accept = {
		custom_description = {
			text = "spending_hook"
			subject = scope:actor
			object = scope:recipient
			scope:hook = yes
		}
	}	

	on_accept = {
		scope:recipient = {
			every_courtier_or_guest = {
				limit = { is_female = no is_knight = yes }
				had_sex_with_effect = { CHARACTER = scope:recipient PREGNANCY_CHANCE = pregnancy_chance }
				had_sex_with_effect = { CHARACTER = scope:actor PREGNANCY_CHANCE = 0 } 
				scope:actor = { 
					add_trait_xp = { trait = deviants_succubus track = loyalty value = 1 } 
					add_character_modifier = { modifier = deviants_fuckedsilly months = 1 }
				} 
				scope:recipient = { add_character_modifier = { modifier = deviants_fuckedsilly months = 1 } }
			}
		}
		scope:actor = {
			if = {
				limit = {
					scope:hook = yes
					has_hook = scope:recipient
				}
				use_hook = scope:recipient
			}
			remove_character_modifier = deviants_prostitute
			remove_character_modifier = deviants_slave
		}
	}
	ai_accept = {
		base = 0
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient
			multiplier = 1
		}
	}
	ai_frequency = 2
	ai_targets = { 	
		ai_recipients = liege
		ai_recipients = vassals
		ai_recipients = peer_vassals
		ai_recipients = neighboring_rulers
	}
	ai_potential = { always = yes }
	ai_will_do = { 
		base = 50 
		modifier = {
			add = 50
			scope:recipient = {
				is_ai = no
			}
		}
	}
}
