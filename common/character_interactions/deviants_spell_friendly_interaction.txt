deviants_weight_interaction = {
	icon = d_full_body
	desc = deviants_weight_interaction.desc
    	category = interaction_category_friendly
    	use_diplomatic_range = yes
	common_interaction = yes
	cooldown = { years = 1 }

	is_shown = {
		scope:actor = {
			OR = {
				has_trait_xp = {
					trait = deviants_succubus
					track = power
					value >= 2
				}
				has_trait_xp = {
					trait = deviants_lesser_succubus
					value >= 4
				}
			}
		}
	}
	on_accept = {
		scope:recipient = {
			change_current_weight = -200
			change_current_weight = 100
		}
		scope:actor = {
			if = {
				limit = { has_trait = deviants_succubus }
				add_trait_xp = {
					trait = deviants_succubus
					track = power
					value = -2
				}
			}
			if = {
				limit = { has_trait = deviants_lesser_succubus }
				add_trait_xp = {
					trait = deviants_lesser_succubus
					value = -4
				}
			}
		}
	}
	auto_accept = yes
	ai_frequency = 36
	ai_targets = { ai_recipients = self }
	ai_potential = { always = yes }
	ai_will_do = { 
		base = 0 
		modifier = {
			base = 100
			scope:recipient = scope:actor
		}
	}
}

deviants_befriend_interaction = {
	icon = synergise_friendship_interaction
	desc = deviants_befriend_interaction.desc
    	category = interaction_category_friendly
    	use_diplomatic_range = yes
	cooldown = { years = 1 }
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = { 
			NOT = { has_relation_friend = scope:recipient } 
			has_trait_xp = {
				trait = deviants_succubus
				track = power
				value >= 10
			} 
		}
	}
	on_accept = {
		scope:actor = {
			set_relation_friend = scope:recipient
			add_trait_xp = {
				trait = deviants_succubus
				track = power
				value = -10
			}
		}
	}
	auto_accept = yes
}

deviants_rival_interaction = {
	icon = icon_hostile
	desc = deviants_rival_interaction.desc
    	category = interaction_category_friendly
    	use_diplomatic_range = yes
	cooldown = { years = 1 }
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = { 
			OR = {
				has_relation_rival = scope:recipient
				has_relation_nemesis = scope:recipient
			}
			has_trait_xp = {
				trait = deviants_succubus
				track = power
				value >= 10
			} 
		}
	}
	on_accept = {
		scope:actor = {
			if = {
				limit = { has_relation_rival = scope:recipient }
				remove_relation_rival = scope:recipient
			}
			if = {
				limit = { has_relation_nemesis = scope:recipient }
				remove_relation_nemesis = scope:recipient
			}
			add_trait_xp = {
				trait = deviants_succubus
				track = power
				value = -10
			}
		}
	}
	auto_accept = yes
	ai_frequency = 12
	ai_targets = { ai_recipients = scripted_relations }
	ai_potential = { always = yes }
	ai_will_do = { base = 100 }
}
