deviants_add_courtier_interaction = {
	icon = invite_to_court_interaction
	desc = deviants_add_courtier_interaction.desc
    	category = interaction_category_vassal
    	use_diplomatic_range = yes
	common_interaction = yes
	cooldown = { years = 1 }
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = { 
			has_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value >= 1
			} 
		}
		scope:recipient = { 
			OR = {
				is_courtier = yes
				NOT = { exists = scope:recipient.liege }
			}
			NOT = { is_courtier_of = scope:actor }
		}
	}
	on_accept = {
		scope:actor = {
			add_courtier = scope:recipient
			add_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value = -1
			}
		}
		scope:recipient = {
			if = {
				limit = { 
					is_female = yes
					age >= 16 
					scope:actor = {
						any_courtier = {
							has_character_modifier = deviants_prostitute
							count < 3
						}
					}
				}
				add_character_modifier = deviants_prostitute
			}
		}
	}
	auto_accept = yes
	ai_frequency = 12
	ai_targets = { 
		ai_recipients = courtiers 
		ai_recipients = prisoners
		ai_recipients = vassals
		ai_recipients = self
	}
	ai_potential = { always = yes }
	ai_will_do = {
		base = 0
		modifier = {
			add = 50
			scope:actor = {
				any_courtier = {
					has_character_modifier = deviants_prostitute
					count < 3
				}
			}
		}
	}
}

deviants_offer_devils_virgin = {
	icon = d_virgin
	desc = deviants_offer_devils_virgin.desc
    	category = interaction_category_diplomacy
	is_shown = {
		scope:actor = {
			is_female = yes
			NOR = { 
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				has_trait = deviants_virgin
			}
			has_trait = lustful
			age <= 30
		}
		scope:recipient = {
			has_trait = deviants_succubus 
		}
	}
	on_accept = {
		scope:actor = {
			add_trait = deviants_virgin
			add_secret = { 
				type = secret_deviant
			}
			add_secret = { 
				type = secret_witch
			}
		}
	}

	ai_accept = {
		base = 50
		modifier = {
			add = 100
			scope:recipient = { has_trait = deviants_lesser_succubus }
		}
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient
			multiplier = 1
		}
	}
	ai_frequency = 12
	ai_targets = {
		ai_recipients = liege
		ai_recipients = vassals
		ai_recipients = peer_vassals
	}
	ai_potential = { 
		primary_title.tier >= tier_county
	}
	ai_will_do = {
		base = 5
	}
}
