			is_adult = yes

		scope:recipient = {

				limit = { matrilinear_marriage = yes }
				create_character = {
					age = 0
					mother = scope:recipient

				}
			}
			else = {
				create_character = {
					age = 0
					mother = scope:recipient
					father = scope:recipient.primary_spouse

			}
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value = -25
			}
		}
	}
	auto_accept = yes
	ai_frequency = 12
	ai_targets = { ai_recipients = self }
	ai_potential = { always = yes }
	ai_will_do = { 
		base = 0
		modifier = {
			add = 20
