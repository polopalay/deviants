deviants_switch_body = {
	icon = d_switch
	desc = deviants_switch_body.desc
    	category = interaction_category_diplomacy
	interface_priority = 1
	common_interaction = yes
    	use_diplomatic_range = yes
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = { is_ai = no has_trait = deviants_succubus }
		scope:recipient = { has_trait = deviants_succubus }
	}
	on_accept = {
		scope:actor = {
			set_player_character = scope:recipient
		}
	}
	auto_accept = yes
}

deviants_add_courtier = {
	icon = d_invite_to_court
	desc = deviants_add_courtier.desc
    	category = interaction_category_vassal
    	use_diplomatic_range = yes
	common_interaction = yes
	cooldown = { years = 1 }
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = { 
			has_trait_xp = {
				trait = deviants_succubus
				track = power
				value >= 10
			} 
		}
		scope:recipient = { 
			OR = {
				is_courtier = yes
				NOT = { exists = scope:recipient.liege }
			}
			NOT = { is_courtier_of = scope:actor }
		}
	}
	on_accept = {
		scope:actor = {
			add_courtier = scope:recipient
			add_trait_xp = {
				trait = deviants_succubus
				track = power
				value = -10
			}
		}
	}
	auto_accept = yes
}

			}

							faith = primary_spouse.faith
							gender = female
						}
					}
					else = {
						if = {
							limit = { exists = primary_spouse }
							create_character = {
								age = 0
								mother = scope:recipient
								father = primary_spouse
								dynasty_house = house
								culture = culture

								gender = female
							}

						else = {
