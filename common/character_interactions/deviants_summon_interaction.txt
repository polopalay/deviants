deviants_switch_body = {
	icon = deviantsfashion
	desc = deviants_switch_body.desc
    	category = interaction_category_diplomacy
    	use_diplomatic_range = yes
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = { is_ai = no has_trait = deviants_succubus }
		scope:recipient = { has_trait = deviants_succubus }
	}
	on_accept = {
		scope:actor = {
			set_player_character = scope:recipient
		}
	}
	auto_accept = yes
}

deviants_child_interaction = {
	icon = deviantsfamily
	desc = deviants_child_interaction.desc
    	category = interaction_category_vassal
    	use_diplomatic_range = yes
	cooldown = { years = 10 }
	is_shown = {
		scope:actor = {
			is_female = yes
			is_adult = yes
			has_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value >= 25
			} 
		}
		scope:recipient = {
			is_adult = yes
			is_female = yes
		}
	}
	on_accept = {
		scope:recipient = {
			if = {
				limit = { matrilinear_marriage = yes }
				create_character = {
					age = 0
					mother = scope:recipient
					father = scope:recipient.primary_spouse
					dynasty_house = scope:recipient.house
				}
			}
			else = {
				create_character = {
					age = 0
					mother = scope:recipient
					father = scope:recipient.primary_spouse
					dynasty_house = scope:recipient.primary_spouse.house
				}
			}
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value = -25
			}
		}
	}
	auto_accept = yes
	ai_frequency = 120
	ai_targets = { ai_recipients = self }
	ai_potential = { always = yes }
	ai_will_do = { 
		base = 0
		modifier = {
			add = 100
			scope:recipient = {
				any_child = { count = 0 }
			}
			scope:actor = scope:recipient
		}
		modifier = {
			add = 20
			scope:recipient = {
				any_child = {
					is_alive = yes
					count =0
				}
			}
			scope:actor = scope:recipient
		}
	}
}
