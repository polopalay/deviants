deviants_summon_legendary_beast_interaction = {
	icon = deviantsbeast
	desc = deviants_summon_legendary_beast_interaction.desc
    	category = interaction_category_vassal
    	use_diplomatic_range = yes
	cooldown = { years = 2 }
	is_shown = {
		scope:actor = scope:recipient
		scope:actor = {
			is_ruler = yes
			OR = {
				has_trait_xp = {
					trait = deviants_succubus
					track = demonanalcorruption
					value >= 10
				}
			}
		}
		scope:recipient.capital_county = { NOT = { has_county_modifier = hunt_sighting_legendary_modifier } }
	}
	on_accept = {
		scope:recipient.capital_county = { 
			add_county_modifier = {
				modifier = hunt_sighting_legendary_modifier 
				years = 2
			}
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = demonanalcorruption
				value = -10
			}
		}
	}
	auto_accept = yes
	ai_frequency = 12
	ai_targets = { ai_recipients = self }
	ai_potential = { always = yes }
	ai_will_do = { base = 5 }
}

deviants_child_interaction = {
	icon = deviantsbastard
	desc = deviants_child_interaction.desc
    	category = interaction_category_vassal
    	use_diplomatic_range = yes
	cooldown = { years = 10 }
	is_shown = {
		scope:recipient = scope:actor
		scope:actor = {
			is_female = yes
			is_adult = yes
			has_trait_xp = {
				trait = deviants_succubus
				track = demonanalcorruption
				value >= 25
			} 
		}
		scope:recipient = {
			exists = scope:recipient.primary_spouse
			NOT = {
				any_child = {
					is_alive = yes
					count >= 2
				}
			}
		}
	}
	on_accept = {
		create_character = {
			age = 0
			mother = scope:recipient
			father = scope:recipient.primary_spouse
			dynasty_house = scope:recipient.house
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = demonanalcorruption
				value = -25
			}
		}
	}
	auto_accept = yes
	ai_frequency = 120
	ai_targets = { ai_recipients = self }
	ai_potential = { always = yes }
	ai_will_do = { 
		base = 5
		modifier = {
			add = 45
			any_child = { 
				is_alive = yes
				count = 0
			}
		}
	}
}

deviants_create_lesser_succubus_interaction = {
	icon = titsgood
	desc = deviants_spawn_darkarmy_interaction.desc
    	category = interaction_category_vassal
    	use_diplomatic_range = yes
	cooldown = { years = 10 }
	is_shown = {
		scope:recipient = {
			is_courtier_of = scope:actor
			is_female = yes
			NOR = { 
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
			}
		}
		scope:actor = {
			has_trait_xp = {
				trait = deviants_succubus
				track = demonanalcorruption
				value >= 50
			} 
		}
	}
	on_accept = {
		scope:recipient = {
			if = {
				limit = { has_trait = chaste }
				remove_trait = chaste
			}
			add_trait = lustful
			add_trait = deviants_lesser_succubus
			add_character_modifier = deviants_prostitute
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = demonanalcorruption
				value = -50
			}
		}
	}
	auto_accept = yes
	ai_frequency = 120
	ai_targets = { ai_recipients = self }
	ai_potential = { always = yes }
	ai_will_do = { base = 100 }
}
