succubus_weight_interaction = {
	icon = full_body_hair
	desc = succubus_weight_interaction.desc
	category = interaction_category_friendly
	use_diplomatic_range = yes
	cooldown = { years = 1 }

	is_shown = {
		scope:actor = {
			OR = {
				has_trait = demon
				has_trait_xp = {
					trait = succubus
					track = daring
					value >= 2
				}
				has_trait_xp = {
					trait = lesser_succubus
					value >= 4
				}
			}
		}
		scope:recipient = {
			OR = {
				current_weight >= 25
				current_weight <= -25
			}
		}
	}
	on_accept = {
		scope:recipient = {
			change_current_weight = -200
			change_current_weight = 100
		}
		scope:actor = {
			if = {
				limit = { has_trait = succubus }
				add_trait_xp = {
					trait = succubus
					track = daring
					value = -2
				}
			}
			if = {
				limit = { has_trait = lesser_succubus }
				add_trait_xp = {
					trait = lesser_succubus
					value = -4
				}
			}
		}
	}
	ignores_pending_interaction_block = yes
	auto_accept = yes
	ai_frequency = 12
	ai_targets = { ai_recipients = self }
	ai_potential = { always = yes }
	ai_will_do = { base = 100 }
}

succubus_honor_interaction = {
	icon = rumor
	desc = succubus_honor_interaction.desc
	category = interaction_category_friendly
	use_diplomatic_range = yes
	cooldown = { years = 5 }
	is_shown = {
		scope:actor = {
			OR = {
				has_trait = demon
				has_trait_xp = {
					trait = succubus
					track = daring
					value >= 5
				}
				has_trait_xp = {
					trait = lesser_succubus
					value >= 10
				}
			}
		}
		scope:recipient = {
			OR = {
				has_trait = bastard
				has_trait = legitimized_bastard
				has_trait = disputed_heritage
				has_trait = child_of_concubine_female
				has_trait = child_of_concubine_male
				has_trait = wild_oat
				has_trait = bastard_founder
				has_trait = witch
				has_trait = deviant
				has_trait = cannibal
				has_trait = deviant
				has_trait = adulterer
				has_trait = sodomite
				has_trait = incestuous
				has_trait = fornicator
				has_trait = kinslayer_1
				has_trait = kinslayer_2
				has_trait = kinslayer_3
				has_trait = murderer
			}
		}
	}
	on_accept = {
		scope:recipient = {
			if = {
				limit = { NOT = { scope:actor = scope:recipient } }
				add_opinion = {
					target = scope:actor
					modifier = be_healed_opinion
				}
			}
			remove_trait = bastard
			remove_trait = legitimized_bastard
			remove_trait = disputed_heritage
			remove_trait = child_of_concubine_female
			remove_trait = child_of_concubine_male
			remove_trait = wild_oat
			remove_trait = bastard_founder
			if = {
				limit = { has_trait = witch }
				remove_trait = witch
				add_secret = { type = secret_witch }
			}
			if = {
				limit = { has_trait = deviant }
				remove_trait = deviant
				add_secret = { type = secret_deviant }
			}
			if = {
				limit = { has_trait = cannibal }
				remove_trait = cannibal
				add_secret = { type = secret_cannibal }
			}
			remove_trait = adulterer
			remove_trait = sodomite
			remove_trait = incestuous
			remove_trait = fornicator
			remove_trait = kinslayer_1
			remove_trait = kinslayer_2
			remove_trait = kinslayer_3
			remove_trait = murderer
		}
		scope:actor = {
			if = {
				limit = { has_trait = succubus }
				add_trait_xp = { trait = succubus track = daring value = -5 }
			}
			if = {
				limit = { has_trait = lesser_succubus }
				add_trait_xp = { trait = lesser_succubus value = -10 }
			}
		}
	}
	ignores_pending_interaction_block = yes
	auto_accept = yes
	ai_frequency = 1
	ai_targets = { ai_recipients = self }
	ai_potential = { always = yes }
	ai_will_do = {
		base = 0
		modifier = { add = 25 scope:recipient = scope:actor }
	}
}
