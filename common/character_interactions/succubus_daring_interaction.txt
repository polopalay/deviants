succubus_weight_interaction = {
	icon = full_body_hair
	desc = succubus_weight_interaction.desc
	category = interaction_category_friendly
	use_diplomatic_range = yes
	cooldown = { years = 1 }

	is_shown = {
		scope:actor = {
			OR = {
				has_trait_xp = {
					trait = succubus
					track = daring
					value >= 2
				}
				has_trait_xp = {
					trait = lesser_succubus
					value >= 4
				}
			}
		}
		scope:recipient = {
			OR = {
				current_weight >= 25
				current_weight <= -25
			}
		}
	}
	on_accept = {
		scope:recipient = {
			change_current_weight = -200
			change_current_weight = 100
		}
		scope:actor = {
			if = {
				limit = { has_trait = succubus }
				add_trait_xp = {
					trait = succubus
					track = daring
					value = -2
				}
			}
			if = {
				limit = { has_trait = lesser_succubus }
				add_trait_xp = {
					trait = lesser_succubus
					value = -4
				}
			}
		}
	}
	auto_accept = yes
	ai_frequency = 12
	ai_targets = { ai_recipients = self }
	ai_potential = { always = yes }
	ai_will_do = { base = 100 }
}

succubus_honor_interaction = {
	icon = rumor
	desc = succubus_honor_interaction.desc
	category = interaction_category_friendly
	use_diplomatic_range = yes
	is_shown = {
		scope:actor = {
			OR = {
				has_trait_xp = {
					trait = succubus
					track = daring
					value >= 5
				}
				has_trait_xp = {
					trait = lesser_succubus
					value >= 10
				}
			}
		}
		scope:recipient = {
			OR = {
				has_trait = witch
				has_trait = deviant
				has_trait = cannibal
				has_trait = deviant
				has_trait = adulterer
				has_trait = sodomite
				has_trait = incestuous
				has_trait = fornicator
				has_trait = kinslayer_1
				has_trait = kinslayer_2
				has_trait = kinslayer_3
				has_trait = murderer
			}
		}
	}
	on_accept = {
		scope:recipient = {
			if = {
				limit = { has_trait = witch }
				remove_trait = witch
				add_secret = { type = secret_witch }
			}
			if = {
				limit = { has_trait = deviant }
				remove_trait = deviant
				add_secret = { type = secret_deviant }
			}
			if = {
				limit = { has_trait = cannibal }
				remove_trait = cannibal
				add_secret = { type = secret_cannibal }
			}
			remove_trait = adulterer
			remove_trait = sodomite
			remove_trait = incestuous
			remove_trait = fornicator
			remove_trait = kinslayer_1
			remove_trait = kinslayer_2
			remove_trait = kinslayer_3
			remove_trait = murderer
		}
		scope:actor = {
			if = {
				limit = { has_trait = succubus }
				add_trait_xp = { trait = succubus track = daring value = -5 }
			}
			if = {
				limit = { has_trait = lesser_succubus }
				add_trait_xp = { trait = lesser_succubus value = -10 }
			}
		}
	}
	auto_accept = yes
}

succubus_mind_break_interaction = {
	icon = mesmerize
	desc = succubus_mind_break_interaction.desc
	category = interaction_category_hostile
	use_diplomatic_range = yes
	cooldown = { years = 1 }
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = {
			has_trait_xp = {
				trait = succubus
				track = daring
				value >= 50
			}
			is_valid_construction_alchemy2_trigger = yes
		}
		scope:recipient = {
			is_female = yes
			age >= 16
			age <= 30
			NOR = {
				has_trait = succubus
				has_trait = lesser_succubus
				has_trait = deviant_virgin
			}
		}
	}
	on_accept = {
		scope:actor = {
			add_trait_xp = {
				trait = succubus
				track = daring
				value = -50
			}
			add_trait_xp = {
				trait = succubus
				track = loyalty
				value = 1
			}
		}
		scope:recipient = {
			if = {
				limit = { has_trait = chaste }
				remove_trait = chaste
			}
			if = {
				limit = { have_devil_nicknames_trigger = no }
				give_prostitution_nickname = yes
			}
			add_trait = lustful
			add_trait = deviant_virgin
			add_character_modifier = diplomacy_majesty_4090_no_raiment_modifier
			add_secret = { type = secret_deviant }
			add_secret = { type = secret_witch }
		}
	}
	auto_accept = yes
	ai_frequency = 60
	ai_targets = {
		ai_recipients = liege
		ai_recipients = vassals
		ai_recipients = peer_vassals
	}
	ai_potential = { always = yes }
	ai_will_do = { base = 5 }
}
