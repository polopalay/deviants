deviants_switch_body_interaction = {
	icon = two_body
	desc = deviants_switch_body_interaction.desc
    	category = interaction_category_diplomacy
	interface_priority = 1
	common_interaction = yes
    	use_diplomatic_range = no
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = { 
			is_ai = no
			OR = {
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				has_trait = deviants_virgin
			}
		}
		scope:recipient = { 
			is_ruler = yes
			primary_title.tier >= tier_county
			OR = {
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				has_trait = deviants_virgin
			}
		}
	}
	on_accept = {
		scope:actor = { set_player_character = scope:recipient }
	}
	auto_accept = yes
}

deviants_add_courtier_interaction = {
	icon = fascinate
	desc = deviants_add_courtier_interaction.desc
    	category = interaction_category_vassal
    	use_diplomatic_range = yes
	common_interaction = no
	cooldown = { years = 1 }
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = { 
			has_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value >= 1
			} 
		}
		scope:recipient = { 
			OR = {
				is_courtier = yes
				NOT = { exists = scope:recipient.liege }
			}
			NOT = { is_courtier_of = scope:actor }
		}
	}
	on_accept = {
		scope:actor = {
			add_courtier = scope:recipient
			add_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value = -1
			}
		}
		scope:recipient = {
			if = {
				limit = { 
					is_female = yes
					age >= 16 
					scope:actor = {
						any_courtier = {
							has_character_modifier = deviants_prostitute
							count <= 2
						}
					}
				}
				add_character_modifier = deviants_prostitute
				if = {
					limit = { have_devil_nicknames_trigger = no }
					give_prostitution_nickname = yes
				}
			}
		}
	}
	auto_accept = yes
}

deviants_child_interaction = {
	icon = pregnant
	desc = deviants_child_interaction.desc
    	category = interaction_category_friendly
    	use_diplomatic_range = yes
	common_interaction = no
	#cooldown = { years = 2 }
	is_shown = {
		scope:actor = {
			is_female = yes
			is_adult = yes
			has_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value >= 5
			} 
		}
		scope:recipient = {
			is_adult = yes
			is_female = yes
			age >= 18
		}
	}
	on_accept = {
		scope:recipient = {
			if = {
				limit = { patrilinear_marriage = yes }
				create_character = {
					age = 0
					mother = scope:recipient
					father = primary_spouse
					dynasty_house = primary_spouse.house
					culture = primary_spouse.culture
					faith = primary_spouse.faith
					gender_female_chance = 50
					trait = deviants_blood
				}
			}
			else = {
				create_character = {
					age = 0
					mother = scope:recipient
					father = primary_spouse
					dynasty_house = house
					culture = culture
					faith = faith
					gender_female_chance = 50
					trait = deviants_blood
				}
			}
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value = -5
			}
		}
	}
	auto_accept = yes
}

deviants_create_lesser_succubus_interaction = {
	icon = lesser_devil
	desc = deviants_create_lesser_succubus_interaction.desc
    	category = interaction_category_vassal
    	use_diplomatic_range = yes
	cooldown = { years = 10 }
	is_shown = {
		scope:recipient = {
			OR = {
				is_courtier_of = scope:actor
				is_vassal_of = scope:actor
			}
			is_female = yes
			age >= 18
			NOR = { 
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				has_trait = deviants_virgin
			}
		}
		scope:actor = {
			has_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value >= 25
			} 
			is_valid_construction_alchemy3_trigger = yes
		}
	}
	on_accept = {
		scope:recipient = {
			add_secret = { 
				type = secret_deviant
			}
			add_secret = { 
				type = secret_witch
			}
			if = {
				limit = { have_devil_nicknames_trigger = no }
				give_prostitution_nickname = yes
			}
			add_trait = deviants_lesser_succubus
			add_character_modifier = deviants_prostitute
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value = -25
			}
		}
	}
	auto_accept = yes
}
