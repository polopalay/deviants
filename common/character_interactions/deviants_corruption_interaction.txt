deviants_switch_body_interaction = {
	icon = propose_best_friendship_interaction
	desc = deviants_switch_body_interaction.desc
    	category = interaction_category_diplomacy
	interface_priority = 1
	common_interaction = yes
    	use_diplomatic_range = yes
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = { 
			is_ai = no
			OR = {
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				has_trait = deviants_virgin
			}
		}
		scope:recipient = { 
			is_ruler = yes
			primary_title.tier >= tier_county
			OR = {
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				has_trait = deviants_virgin
			}
		}
	}
	on_accept = {
		scope:actor = {
			set_player_character = scope:recipient
		}
	}
	auto_accept = yes
}

deviants_offer_devils_virgin = {
	icon = sub_chastity_rules_interaction
	desc = deviants_offer_devils_virgin.desc
    	category = interaction_category_diplomacy
	is_shown = {
		scope:actor = {
			is_female = yes
			NOR = { 
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				has_trait = deviants_virgin
			}
			has_trait = lustful
			age <= 30
		}
		scope:recipient = {
			has_trait = deviants_succubus 
		}
	}
	on_accept = {
		scope:actor = {
			add_trait = deviants_virgin
			if = {
				limit = { has_character_modifier = deviants_prostitute }
				remove_character_modifier = deviants_prostitute
			}
			add_character_modifier = deviants_prostitute
			add_character_modifier = diplomacy_majesty_4090_no_raiment_modifier
			add_secret = { 
				type = secret_deviant
			}
			add_secret = { 
				type = secret_witch
			}
		}
	}

	ai_accept = {
		base = 50
		modifier = {
			add = 100
			scope:recipient = { has_trait = deviants_lesser_succubus }
		}
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient
			multiplier = 1
		}
	}
	ai_frequency = 12
	ai_targets = {
		ai_recipients = liege
		ai_recipients = vassals
		ai_recipients = peer_vassals
	}
	ai_potential = { 
		primary_title.tier >= tier_county
	}
	ai_will_do = {
		base = 5
	}
}

deviants_add_courtier_interaction = {
	icon = invite_to_court_interaction
	desc = deviants_add_courtier_interaction.desc
    	category = interaction_category_vassal
    	use_diplomatic_range = yes
	common_interaction = yes
	cooldown = { years = 1 }
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = { 
			has_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value >= 2
			} 
		}
		scope:recipient = { 
			OR = {
				is_courtier = yes
				NOT = { exists = scope:recipient.liege }
			}
			NOT = { is_courtier_of = scope:actor }
		}
	}
	on_accept = {
		scope:actor = {
			add_courtier = scope:recipient
			add_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value = -2
			}
		}
		scope:recipient = {
			if = {
				limit = { 
					is_female = yes
					age >= 16 
					scope:actor = {
						any_courtier = {
							has_character_modifier = deviants_prostitute
							count <= 2
						}
					}
				}
				add_character_modifier = deviants_prostitute
			}
		}
	}
	auto_accept = yes
}

deviants_child_interaction = {
	icon = pregnant
	desc = deviants_child_interaction.desc
    	category = interaction_category_vassal
    	use_diplomatic_range = yes
	common_interaction = yes
	#cooldown = { years = 2 }
	is_shown = {
		scope:actor = {
			is_female = yes
			is_adult = yes
			has_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value >= 5
			} 
		}
		scope:recipient = {
			is_adult = yes
			is_female = yes
			age >= 18
		}
	}
	on_accept = {
		scope:recipient = {
			if = {
				limit = { patrilinear_marriage = yes }
				create_character = {
					age = 0
					mother = scope:recipient
					father = primary_spouse
					dynasty_house = primary_spouse.house
					culture = primary_spouse.culture
					faith = primary_spouse.faith
					gender_female_chance = 50
					trait = deviants_blood
				}
			}
			else = {
				create_character = {
					age = 0
					mother = scope:recipient
					father = primary_spouse
					dynasty_house = house
					culture = culture
					faith = faith
					gender_female_chance = 50
					trait = deviants_blood
				}
			}
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = corruption
				value = -5
			}
		}
	}
	auto_accept = yes
}
