deviants_weight_interaction = {
	icon = full_body_hair
	desc = deviants_weight_interaction.desc
    	category = interaction_category_friendly
    	use_diplomatic_range = yes
	cooldown = { years = 1 }

	is_shown = {
		scope:actor = {
			OR = {
				has_trait_xp = {
					trait = deviants_succubus
					track = daring
					value >= 2
				}
				has_trait_xp = {
					trait = deviants_lesser_succubus
					value >= 4
				}
			}
		}
		scope:recipient = {
			OR = {
				current_weight >= 25
				current_weight <= -25
			}
		}
	}
	on_accept = {
		scope:recipient = {
			change_current_weight = -200
			change_current_weight = 100
		}
		scope:actor = {
			if = {
				limit = { has_trait = deviants_succubus }
				add_trait_xp = {
					trait = deviants_succubus
					track = daring
					value = -2
				}
			}
			if = {
				limit = { has_trait = deviants_lesser_succubus }
				add_trait_xp = {
					trait = deviants_lesser_succubus
					value = -4
				}
			}
		}
	}
	auto_accept = yes
	ai_frequency = 12
	ai_targets = { ai_recipients = self }
	ai_potential = { always = yes }
	ai_will_do = { base = 100 }
}

deviants_mind_break_interaction = {
	icon = mesmerize
	desc = deviants_mind_break_interaction.desc
    	category = interaction_category_hostile
    	use_diplomatic_range = yes
	cooldown = { years = 1 }
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = { 
			has_trait_xp = {
				trait = deviants_succubus
				track = daring
				value >= 50
			} 
			is_valid_construction_alchemy2_trigger = yes
		}
		scope:recipient = {
			is_female = yes
			age >= 16
			age <= 30
			NOR = {
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				has_trait = deviants_virgin
			}
		}
	}
	on_accept = {
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = daring
				value = -50
			}
			add_trait_xp = {
				trait = deviants_succubus
				track = loyalty
				value = 1
			}
		}
		scope:recipient = {
			if = {
				limit = { has_character_modifier = deviants_prostitute }
				remove_character_modifier = deviants_prostitute
			}
			if = {
				limit = { has_trait = chaste }
				remove_trait = chaste
			}
			if = {
				limit = { have_devil_nicknames_trigger = no }
				give_prostitution_nickname = yes
			}
			add_trait = lustful
			add_trait = deviants_virgin
			add_character_modifier = deviants_prostitute
			add_character_modifier = diplomacy_majesty_4090_no_raiment_modifier
			add_secret = { type = secret_deviant }
			add_secret = { type = secret_witch }
		}
	}
	auto_accept = yes
	ai_frequency = 60
	ai_targets = {
		ai_recipients = liege
		ai_recipients = vassals
		ai_recipients = peer_vassals
	}
	ai_potential = { always = yes }
	ai_will_do = { base = 5 }
}
