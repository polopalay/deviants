			OR = {
				has_trait = sbs_demon

			}

			NOT = {
				has_trait = seed_of_destruction

		}

			}
		}

						limit = { is_ruler = no has_trait = deviant_blood }
						remove_trait = deviant_blood
						create_landless_adventurer_title_effect = {
							REASON = flag:voluntary
							FLAVOR_CHAR = scope:recipient
						}
					}
				}
				5 = {
					modifier = { has_trait = deviant_blood add = 5 }
					add_character_modifier = succubus_debuff
				}
				5 = {
					modifier = { any_secret = { secret_type = secret_deviant } add = -5 }
					scope:recipient = {
						add_secret = { type = secret_deviant }
					}
				}
				5 = {
					modifier = { any_secret = { secret_type = secret_witch } add = -5 }
					scope:recipient = { add_secret = { type = secret_witch } }
				}
				5 = {
					modifier = {
						NOR = {
							has_trait = slave
							has_character_flag = carn_cannot_be_enslaved
							has_trait_with_flag = carn_cannot_be_enslaved
							has_opinion_modifier = { target = scope:actor modifier = carn_wants_to_be_your_slave_opinion }
						}
						add = -5
					}
					add_opinion = { target = scope:actor modifier = carn_wants_to_be_your_slave_opinion }
				}
				5 = {
					pay_short_term_gold = {
						target = scope:actor
						gold = tiny_gold_value
					}
				}
				5 = { add_trait = depressed_1 }
				5 = { add_trait = depressed_genetic }
				5 = { add_trait = lunatic_1 }
				5 = { add_trait = lunatic_genetic }

				5 = { add_trait = possessed_genetic }
				5 = { add_trait = ill }

				5 = { add_trait = gout_ridden }
				5 = { add_trait = typhus }
				5 = { add_trait = consumption }
				5 = { add_trait = cancer }

				5 = { add_trait = ergotism }
