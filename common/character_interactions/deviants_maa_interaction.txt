deviants_absorb_interaction = {
	icon = d_loyalty
	desc = deviants_absorb_interaction.desc
    	category = interaction_category_vassal
    	use_diplomatic_range = yes
	common_interaction = yes
	cooldown = { years = 4 }
	is_shown = {
		scope:actor = {
			has_trait = deviants_succubus
			is_ruler = yes
			NOT = { is_at_war_with = scope:recipient }
			every_courtier_or_guest = {
				limit = { is_female = no is_knight = yes }
				count > 0
			}
		}
		scope:recipient = {
			OR = {
				is_courtier_of = scope:actor
				is_vassal_of = scope:actor
			}
			has_trait_xp = { trait = deviants_lesser_succubus value >= 50 }
		}
	}
	is_valid = {
		scope:actor = { is_valid_construction_brothel2_trigger = yes }
	}
	on_accept = {
		scope:recipient = {
			add_trait_xp = {
				trait = deviants_lesser_succubus
				value = -50
			}
		}
		scope:actor = {
			add_trait_xp = {
				trait = deviants_succubus
				track = loyalty
				value = 5
			}
		}
	}
	auto_accept = yes
	ai_frequency = 60
	ai_targets = { 
		ai_recipients = courtiers 
		ai_recipients = prisoners
		ai_recipients = vassals
	}
	ai_potential = { always = yes }
	ai_will_do = { base = 100 }
}

deviants_dispel_interaction = {
	icon = d_drain_life
	desc = deviants_dispel_interaction.desc
    	category = interaction_category_diplomacy
    	use_diplomatic_range = yes
	common_interaction = yes
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			NOT = { is_at_war_with = scope:recipient }
			is_adult = yes
			is_ruler = yes
			is_female = yes
			OR = {
				is_ruler = yes
				NOT = { is_courtier = scope:recipient }
			}
			NOT  = { 
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				is_vassal_of = scope:recipient
				any_relation = { type = deviants_slaver }
				any_relation = { type = deviants_partner }
				scope:actor.liege = {
					is_valid_construction_brothel1_trigger	= yes
				}
			}
			OR = {
				has_character_modifier = diplomacy_majesty_4090_no_raiment_modifier
			}
		}
		scope:recipient = {
			is_ruler = yes
			has_trait = deviants_succubus 
		}
	}
	send_option = {
		is_valid = { scope:actor = { has_usable_hook = scope:recipient } }
		flag = hook	
		localization = GENERIC_SPEND_A_HOOK
	}
	send_options_exclusive = no
	auto_accept = {
		custom_description = {
			text = "spending_hook"
			subject = scope:actor
			object = scope:recipient
			scope:hook = yes
		}
	}	

	on_accept = {
		scope:actor = {
			if = {
				limit = {
					scope:hook = yes
					has_hook = scope:recipient
				}
				use_hook = scope:recipient
			}
			if = {
				limit = { has_character_modifier = diplomacy_majesty_4090_no_raiment_modifier }
				scope:recipient = {
					random_courtier = {
						limit = { is_female = no is_adult = yes is_knight = yes }
						had_sex_with_effect = { CHARACTER = scope:actor PREGNANCY_CHANCE = pregnancy_chance } 
						scope:actor = { 
							remove_character_modifier = diplomacy_majesty_4090_no_raiment_modifier
							add_character_modifier = { modifier = deviants_fuckedsilly months = 1 }
						} 
						had_sex_with_effect = { CHARACTER = scope:recipient PREGNANCY_CHANCE = pregnancy_chance }
						scope:recipient = { 
							add_character_modifier = { modifier = deviants_fuckedsilly months = 1 }
							random = {
								chance = 50
								add_trait_xp = { trait = deviants_succubus track = loyalty value = 1 } 
							}
							add_trait_xp = { trait = deviants_succubus track = power value = 1 }
						} 
					}
				}
			}
		}
	}
	ai_accept = {
		base = -10
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient
			multiplier = 1
		}
	}
	ai_frequency = 1
	ai_targets = {
		ai_recipients = liege
		ai_recipients = vassals
		ai_recipients = peer_vassals
	}
	ai_potential = { always = yes }
	ai_will_do = { 
		base = 50
	}
}

	common_interaction = yes
	send_options_exclusive = no
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			NOT = { is_at_war_with = scope:recipient }
			is_adult = yes
			is_ruler = yes
			is_female = yes
			OR = {
				age > 36
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
			}
		}
		scope:recipient = {
			OR = {
				has_relation_deviants_slave_owner = scope:actor
				has_relation_deviants_partner = scope:actor
			}
			is_valid_construction_brothel2_trigger = yes
		}
	}
	send_option = {
		is_valid = { scope:actor = { has_usable_hook = scope:recipient } }
		flag = hook	
		localization = GENERIC_SPEND_A_HOOK
	}
	auto_accept = {
		custom_description = {
			text = "spending_hook"
			subject = scope:actor
			object = scope:recipient
			scope:hook = yes
		}
	}	

	on_accept = {
		scope:actor = {
			if = {
				limit = { scope:hook = yes has_hook = scope:recipient }
				use_hook = scope:recipient
			}
			if = {
				limit = { has_character_modifier = deviants_slave }
				scope:recipient = {
					every_courtier = {
						limit = { is_female = no is_adult = yes }

							remove_character_modifier = deviants_slave
							add_character_modifier = { modifier = deviants_fuckedsilly months = 1 }

						had_sex_with_effect = { CHARACTER = scope:recipient PREGNANCY_CHANCE = pregnancy_chance }
						scope:recipient = { 
							add_character_modifier = { modifier = deviants_fuckedsilly months = 1 }
							random = {
								chance = 25
								add_trait_xp = { trait = deviants_succubus track = corruption value = 1 } 
							}
							add_trait_xp = { trait = deviants_succubus track = power value = 1 }
						} 
					}
				}
				remove_relation_deviants_slaver = scope:recipient
			}
			if = {
				limit = { has_character_modifier = deviants_prostitute }
				scope:recipient = {
					random_courtier = {
						limit = { is_female = no is_adult = yes }
						had_sex_with_effect = { CHARACTER = scope:actor PREGNANCY_CHANCE = pregnancy_chance } 
						scope:actor = { 
							remove_character_modifier = deviants_prostitute
							add_character_modifier = { modifier = deviants_fuckedsilly months = 1 }
						}
						had_sex_with_effect = { CHARACTER = scope:recipient PREGNANCY_CHANCE = pregnancy_chance }
						scope:recipient = { 
							add_character_modifier = { modifier = deviants_fuckedsilly months = 1 }
							random = {
								chance = 25
								add_trait_xp = { trait = deviants_succubus track = loyalty value = 1 } 
							}
							add_trait_xp = { trait = deviants_succubus track = power value = 1 }
						} 
					}
				}
				remove_relation_deviants_partner = scope:recipient
			}
		}
		scope:recipient = {
			remove_relation_deviants_slave_owner = scope:actor
			remove_relation_deviants_partner = scope:actor
		}
	}

		base = -40

					has_trait = deviants_succubus
					has_trait = deviants_lesser_succubus
				}
			}
		}

	ai_frequency = 1
	ai_targets = {
		ai_recipients = scripted_relations
	}
	ai_potential = { always = yes }
	ai_will_do = { 
		base = 5 
		modifier = {
			add = 45
			scope:recipient = {
				age > 40
			}
		}
		modifier = {
			add = 50
			scope:recipient = {
				OR = {
					has_trait = deviants_succubus
					has_trait = deviants_lesser_succubus
				}
			}
