succubus_create_lesser_succubus_interaction = {
	icon = lesser_devil
	desc = succubus_create_lesser_succubus_interaction.desc
	category = interaction_category_vassal
	use_diplomatic_range = yes
	cooldown = { years = 25 }
	is_shown = {
		scope:recipient = {
			NOR = {
				has_trait = demon
				has_trait = succubus
				has_trait = lesser_succubus
			}
		}
		scope:actor = {
			OR = {
				has_trait = demon
				AND = {
					has_trait_xp = { trait = succubus track = corruption value >= 25 }
					is_valid_construction_alchemy3_trigger = yes
				}
			}
		}
	}
	on_accept = {
		scope:recipient = {
			if = {
				limit = { has_trait = deviant_virgin }
				remove_trait = deviant_virgin
			}
			mind_break_personality = yes
			add_trait = lesser_succubus
			set_immortal_age = 18
		}
		scope:actor = {
			if = {
				limit = { has_trait = succubus }
				add_trait_xp = { trait = succubus track = corruption value = -25 }
			}
		}
	}
	auto_accept = yes
}

succubus_create_succubus_interaction = {
	icon = devil
	desc = succubus_create_succubus_interaction.desc
	category = interaction_category_vassal
	use_diplomatic_range = yes
	cooldown = { years = 50 }
	is_shown = {
		scope:recipient = {
			NOR = {
				has_trait = demon
				has_trait = succubus
			}
		}
		scope:actor = {
			has_trait = demon
		}
	}
	on_accept = {
		scope:recipient = {
			if = {
				limit = { has_trait = deviant_virgin }
				remove_trait = deviant_virgin
			}
			if = {
				limit = { has_trait = lesser_succubus }
				remove_trait = lesser_succubus
			}
			mind_break_personality = yes
			add_trait = succubus
			set_immortal_age = 18
		}
	}
	auto_accept = yes
}
