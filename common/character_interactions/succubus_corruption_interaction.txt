succubus_switch_body_interaction = {
	icon = two_body
	desc = succubus_switch_body_interaction.desc
	category = interaction_category_diplomacy
	interface_priority = 1
	common_interaction = yes
	use_diplomatic_range = yes
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = {
			is_ai = no
			OR = {
				has_trait = succubus
				has_trait = lesser_succubus
				has_trait = deviant_virgin
			}
		}
		scope:recipient = {
			is_ruler = yes
			primary_title.tier >= tier_county
			OR = {
				has_trait = succubus
				has_trait = lesser_succubus
				has_trait = deviant_virgin
			}
		}
	}
	on_accept = {
		scope:actor = { set_player_character = scope:recipient }
	}
	auto_accept = yes
}

succubus_add_courtier_interaction = {
	icon = fascinate
	desc = succubus_add_courtier_interaction.desc
	category = interaction_category_vassal
	use_diplomatic_range = yes
	common_interaction = no
	cooldown = { years = 1 }
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = {
			has_trait_xp = {
				trait = succubus
				track = corruption
				value >= 1
			}
		}
		scope:recipient = {
			OR = {
				is_courtier = yes
				NOT = { exists = scope:recipient.liege }
			}
			NOT = { is_courtier_of = scope:actor }
		}
	}
	on_accept = {
		scope:actor = {
			add_courtier = scope:recipient
			add_trait_xp = {
				trait = succubus
				track = corruption
				value = -1
			}
		}
	}
	auto_accept = yes
}

succubus_child_interaction = {
	icon = pregnant
	desc = succubus_child_interaction.desc
	category = interaction_category_friendly
	use_diplomatic_range = yes
	common_interaction = no
	cooldown = { years = 2 }
	is_shown = {
		scope:actor = {
			is_female = yes
			is_adult = yes
			has_trait_xp = { trait = succubus track = corruption value >= 5 }
			have_soul_forge_trigger = yes
		}
		scope:recipient = {
			is_adult = yes
			is_female = yes
			age >= 18
		}
	}
	on_accept = {
		scope:recipient = {
			if = {
				limit = { patrilinear_marriage = yes }
				create_character = {
					age = 0
					location = primary_spouse.capital_province
					mother = scope:recipient
					father = primary_spouse
					dynasty_house = primary_spouse.house
					culture = primary_spouse.culture
					faith = primary_spouse.faith
					gender_female_chance = 50
					trait = deviant_blood
				}
			}
			else = {
				create_character = {
					age = 0
					location = capital_province
					mother = scope:recipient
					father = primary_spouse
					dynasty_house = house
					culture = culture
					faith = faith
					gender_female_chance = 50
					trait = deviant_blood
				}
			}
		}
		scope:actor = {
			add_trait_xp = {
				trait = succubus
				track = corruption
				value = -5
			}
		}
	}
	auto_accept = yes
}

succubus_create_lesser_succubus_interaction = {
	icon = lesser_devil
	desc = succubus_create_lesser_succubus_interaction.desc
	category = interaction_category_vassal
	use_diplomatic_range = yes
	cooldown = { years = 10 }
	is_shown = {
		scope:recipient = {
			OR = {
				is_courtier_of = scope:actor
				is_vassal_of = scope:actor
			}
			is_female = yes
			NOR = {
				has_trait = succubus
				has_trait = lesser_succubus
				has_trait = deviant_virgin
			}
		}
		scope:actor = {
			has_trait_xp = {
				trait = succubus
				track = corruption
				value >= 25
			}
			is_valid_construction_alchemy3_trigger = yes
		}
	}
	on_accept = {
		scope:recipient = {
			mind_break_personality = yes
			add_trait = lesser_succubus
			set_immortal_age = 18
		}
		scope:actor = {
			add_trait_xp = {
				trait = succubus
				track = corruption
				value = -25
			}
		}
	}
	auto_accept = yes
}
