deviants_switch_body_interaction = {
	icon = two_body
	desc = deviants_switch_body_interaction.desc
    	category = interaction_category_diplomacy
	interface_priority = 1
	common_interaction = yes
    	use_diplomatic_range = yes
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = { 
			is_ai = no
			OR = {
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				has_trait = deviants_virgin
			}
		}
		scope:recipient = { 
			is_ruler = yes
			primary_title.tier >= tier_county
			OR = {
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				has_trait = deviants_virgin
			}
		}
	}
	on_accept = {
		scope:actor = { set_player_character = scope:recipient }
	}
	auto_accept = yes
}

deviants_start_contraception_interaction = {
	icon = chastity
	desc = deviants_start_contraception_interaction.desc
    	category = interaction_category_friendly
    	use_diplomatic_range = yes

	is_shown = {
		scope:actor = scope:recipient
		scope:actor = { 
			OR = {
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				has_trait = deviants_virgin
			}
			NOT = { has_character_modifier = deviants_contraception }
		}
	}

	on_accept = {
		scope:actor = {
			add_character_modifier = deviants_contraception
		}
	}
}

deviants_stop_contraception_interaction = {
	icon = chastity
	desc = deviants_stop_contraception_interaction.desc
    	category = interaction_category_friendly
    	use_diplomatic_range = yes

	is_shown = {
		scope:actor = scope:recipient
		scope:actor = { 
			OR = {
				has_trait = deviants_succubus
				has_trait = deviants_lesser_succubus
				has_trait = deviants_virgin
			}
			has_character_modifier = deviants_contraception
		}
	}

	on_accept = {
		scope:actor = {
			remove_character_modifier = deviants_contraception
		}
	}
}
