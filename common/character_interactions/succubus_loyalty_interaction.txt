succubus_absorb_interaction = {
	icon = drain_life
	desc = succubus_absorb_interaction.desc
	category = interaction_category_vassal
	use_diplomatic_range = yes
	common_interaction = yes
	cooldown = { months = 1 }
	is_shown = {
		scope:actor = {
			has_trait = succubus
			is_ruler = yes
			NOT = { is_at_war_with = scope:recipient }
			has_trait_xp = { trait = succubus track = loyalty value < 100 }
		}
		scope:recipient = {
			OR = {
				is_courtier_of = scope:actor
				is_vassal_of = scope:actor
			}
			has_trait_xp = { trait = lesser_succubus value >= 50 }
		}
	}
	on_accept = {
		scope:recipient = {
			add_character_modifier = succubus_buff
			add_trait_xp = {
				trait = lesser_succubus
				value = -50
			}
		}
		scope:actor = {
			add_trait_xp = {
				trait = succubus
				track = loyalty
				value = 1
			}
		}
	}
	auto_accept = yes
	ai_frequency = 12
	ai_targets = {
		ai_recipients = courtiers
		ai_recipients = prisoners
		ai_recipients = vassals
	}
	ai_potential = { always = yes }
	ai_will_do = { base = 100 }
}

succubus_cultist_interaction = {
	icon = take_vows
	desc = succubus_cultist_interaction.desc
	category = interaction_category_vassal
	use_diplomatic_range = yes
	common_interaction = no
	cooldown = { years = 1 }

	is_shown = {
		scope:actor = {
			has_trait = succubus
			is_ruler = yes
			is_valid_construction_alchemy1_trigger = yes
		}
		scope:recipient = {
			OR = {
				AND = {
					scope:actor = { has_trait_xp = { trait = succubus track = loyalty value >= 2 } }
					primary_title.tier < tier_county
				}
				AND = {
					scope:actor = { has_trait_xp = { trait = succubus track = loyalty value >= 10 } }
					primary_title.tier = tier_county
				}
				AND = {
					scope:actor = { has_trait_xp = { trait = succubus track = loyalty value >= 25 } }
					primary_title.tier = tier_duchy
				}
				AND = {
					scope:actor = { has_trait_xp = { trait = succubus track = loyalty value >= 50 } }
					primary_title.tier = tier_kingdom
				}
				AND = {
					scope:actor = { has_trait_xp = { trait = succubus track = loyalty value >= 100 } }
					primary_title.tier = tier_empire
				}
			}
		}
	}

	send_option = {
		flag = beelzebub
		localization = beelzebub
	}
	send_option = {
		flag = lilith
		localization = lilith
	}
	send_option = {
		flag = abaddon
		localization = abaddon
	}
	send_option = {
		flag = mammon
		localization = mammon
	}
	send_option = {
		flag = asmodeus
		localization = asmodeus
	}
	send_option = {
		flag = belial
		localization = belial
	}

	on_accept = {
		if = {
			limit = {
				scope:beelzebub = yes
			}
			scope:recipient = {
				set_character_faith_with_conversion = faith:beelzebub
			}
		}
		else_if = {
			limit = {
				scope:lilith = yes
			}
			scope:recipient = {
				set_character_faith_with_conversion = faith:lilith
			}
		}
		else_if = {
			limit = {
				scope:abaddon = yes
			}
			scope:recipient = {
				set_character_faith_with_conversion = faith:abaddon
			}
		}
		else_if = {
			limit = {
				scope:mammon = yes
			}
			scope:recipient = {
				set_character_faith_with_conversion = faith:mammon
			}
		}
		else_if = {
			limit = {
				scope:asmodeus = yes
			}
			scope:recipient = {
				set_character_faith_with_conversion = faith:asmodeus
			}
		}
		else_if = {
			limit = {
				scope:belial = yes
			}
			scope:recipient = {
				set_character_faith_with_conversion = faith:belial
			}
		}

		if = {
			limit = { scope:recipient.primary_title.tier = tier_county }
			scope:actor = {
				add_trait_xp = {
					trait = succubus
					track = loyalty
					value = -10
				}
			}
		}
		else_if = {
			limit = { scope:recipient.primary_title.tier = tier_duchy }
			scope:actor = {
				add_trait_xp = {
					trait = succubus
					track = loyalty
					value = -25
				}
			}
		}
		else_if = {
			limit = { scope:recipient.primary_title.tier = tier_kingdom }
			scope:actor = {
				add_trait_xp = {
					trait = succubus
					track = loyalty
					value = -50
				}
			}
		}
		else_if = {
			limit = { scope:recipient.primary_title.tier = tier_empire }
			scope:actor = {
				add_trait_xp = {
					trait = succubus
					track = loyalty
					value = -100
				}
			}
		}
		else = {
			scope:actor = {
				add_trait_xp = {
					trait = succubus
					track = loyalty
					value = -2
				}
			}
		}
	}

	auto_accept = yes
}
